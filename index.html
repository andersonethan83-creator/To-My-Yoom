<script>

  // Optional: customize question text

  const yesBtn = document.getElementById('yesBtn');

  const noBtn  = document.getElementById('noBtn');

  const buttonsArea = document.getElementById('buttonsArea');

  const overlay = document.getElementById('overlay');

  const openCardBtn = document.getElementById('openCardBtn');

  const cardScreen = document.getElementById('cardScreen');

  const cardText = document.getElementById('cardText');

  const confettiCanvas = document.getElementById('confetti');


  cardText.textContent = CARD_MESSAGE;


  // Keep the NO button inside the stage

  function clamp(v, min, max){ return Math.max(min, Math.min(max, v)); }


  function randomizeNoPosition(avoidX, avoidY){

    const rect = buttonsArea.getBoundingClientRect();

    const pad = 18;


    // button size

    const bRect = noBtn.getBoundingClientRect();

    const bw = bRect.width;

    const bh = bRect.height;


    // Choose a new spot that isn't too close to the cursor

    let x, y, tries = 0;

    do {

      x = pad + Math.random() * (rect.width - pad*2);

      y = pad + Math.random() * (rect.height - pad*2);

      tries++;

      if (tries > 30) break;

    } while (Math.hypot(x - avoidX, y - avoidY) < 140);


    // Clamp within bounds considering button size (because we use translate(-50%, -50%))

    x = clamp(x, bw/2 + pad, rect.width  - bw/2 - pad);

    y = clamp(y, bh/2 + pad, rect.height - bh/2 - pad);


    noBtn.style.left = `${x}px`;

    noBtn.style.top  = `${y}px`;

  }


  // Move NO when mouse gets near

  let noArmed = true;

  buttonsArea.addEventListener('mousemove', (e) => {

    if (!noArmed) return;

    const rect = buttonsArea.getBoundingClientRect();

    const mouseX = e.clientX - rect.left;

    const mouseY = e.clientY - rect.top;


    const b = noBtn.getBoundingClientRect();

    const bx = (b.left - rect.left) + b.width/2;

    const by = (b.top  - rect.top)  + b.height/2;


    const dist = Math.hypot(mouseX - bx, mouseY - by);


    // If near, run away

    if (dist < 120) randomizeNoPosition(mouseX, mouseY);

  });


  // Also dodge on touch devices (when finger approaches)

  buttonsArea.addEventListener('touchstart', (e) => {

    if (!noArmed) return;

    const touch = e.touches[0];

    const rect = buttonsArea.getBoundingClientRect();

    randomizeNoPosition(touch.clientX - rect.left, touch.clientY - rect.top);

  }, { passive:true });


  // If she somehow clicks NO, just move it again

  noBtn.addEventListener('click', (e) => {

    const rect = buttonsArea.getBoundingClientRect();

    randomizeNoPosition(rect.width/2, rect.height/2);

  });


  // YES logic

  yesBtn.addEventListener('click', () => {

    noArmed = false; // stop chasing after YES

    overlay.classList.add('show');

    launchConfetti(1800);

  });


  openCardBtn.addEventListener('click', () => {

  });


  /********************

   * Confetti (no libs)

   ********************/

  function launchConfetti(durationMs){

    confettiCanvas.classList.add('show');


    const ctx = confettiCanvas.getContext('2d');

    const dpr = Math.max(1, window.devicePixelRatio || 1);


    function resize(){

      confettiCanvas.width  = Math.floor(window.innerWidth  * dpr);

      confettiCanvas.height = Math.floor(window.innerHeight * dpr);

      ctx.setTransform(dpr,0,0,dpr,0,0);

    }

    resize();

    window.addEventListener('resize', resize, { passive:true });


    const colors = ['#ffffff', '#ff2a2a', '#ff4d88', '#ffb3c7']; // white, red, pink, light pink

    const pieces = [];

    const count = Math.min(260, Math.floor((window.innerWidth * window.innerHeight) / 9000));


    function rand(min, max){ return min + Math.random()*(max-min); }


    for(let i=0;i<count;i++){

      pieces.push({

        x: rand(0, window.innerWidth),

        y: rand(-window.innerHeight, 0),

        w: rand(6, 12),

        h: rand(8, 16),

        color: colors[Math.floor(Math.random()*colors.length)],

        vx: rand(-2.5, 2.5),

        vy: rand(2.5, 6.5),

        rot: rand(0, Math.PI*2),

        vr: rand(-0.18, 0.18),

        life: rand(0.7, 1.0)

      });

    }


    let start = performance.now();


    function frame(t){

      const elapsed = t - start;

      ctx.clearRect(0,0,window.innerWidth, window.innerHeight);


      for(const p of pieces){

        p.x += p.vx;

        p.y += p.vy;

        p.vy += 0.02;         // gravity

        p.vx *= 0.999;        // slight drag

        p.rot += p.vr;


        // wrap horizontally

        if(p.x < -30) p.x = window.innerWidth + 30;

        if(p.x > window.innerWidth + 30) p.x = -30;


        ctx.save();

        ctx.translate(p.x, p.y);

        ctx.rotate(p.rot);

        ctx.globalAlpha = p.life;

        ctx.fillStyle = p.color;

        ctx.fillRect(-p.w/2, -p.h/2, p.w, p.h);

        ctx.restore();


        // recycle if it falls off bottom during the burst

        if(p.y > window.innerHeight + 40){

          p.y = rand(-120, -20);

          p.x = rand(0, window.innerWidth);

          p.vy = rand(2.5, 6.5);

        }

      }


      if(elapsed < durationMs){

        requestAnimationFrame(frame);

      } else {

        // fade out quickly

        const fadeStart = performance.now();

        function fade(ft){

          const f = Math.min(1, (ft - fadeStart)/350);

          ctx.clearRect(0,0,window.innerWidth, window.innerHeight);

          ctx.globalAlpha = (1 - f);

          for(const p of pieces){

            ctx.save();

            ctx.translate(p.x, p.y);

            ctx.rotate(p.rot);

            ctx.fillStyle = p.color;

            ctx.fillRect(-p.w/2, -p.h/2, p.w, p.h);

            ctx.restore();

          }

          ctx.globalAlpha = 1;

          if(f < 1) requestAnimationFrame(fade);

          else confettiCanvas.classList.remove('show');

        }

        requestAnimationFrame(fade);

      }

    }

    requestAnimationFrame(frame);

  }


  // Start NO in a slightly random spot so it feels alive

  (function initNo(){

    randomizeNoPosition(rect.width * 0.7, rect.height * 0.6);

  })();
